```python
import asyncio
import base64
import uuid
from typing import Optional, Callable, Awaitable

class LocalISP:
    def __init__(self) -> None:
        self._peer: Optional["LocalISP"] = None
        self._on_media: Optional[Callable[[bytes], Awaitable[None]]] = None
        self._on_start: Optional[Callable[[str, str, str], Awaitable[None]]] = None
        self._on_stop: Optional[Callable[[], Awaitable[None]]] = None
        self._queue: asyncio.Queue = asyncio.Queue()
        self._task: Optional[asyncio.Task] = None
        self._inject: Optional[Callable] = None  # Set by conversation loop for event injection

    @classmethod
    def pair(cls, a: "LocalISP", b: "LocalISP") -> None:
        a._peer = b
        b._peer = a

    async def start(self, on_media, on_start, on_stop) -> None:
        self._on_media = on_media
        self._on_start = on_start
        self._on_stop = on_stop
        self._task = asyncio.create_task(self._reader())
        stream_sid = f"local-{uuid.uuid4().hex[:8]}"
        await on_start(stream_sid, "local-call-sid", "local")

    async def _reader(self) -> None:
        while True:
            item = await self._queue.get()
            if item is None:  # sentinel for stop
                break
            await self._on_media(item)

    async def send_audio(self, payload: str) -> None:
        if self._peer is not None:
            audio_bytes = base64.b64decode(payload)
            await self._peer._queue.put(audio_bytes)

    async def send_clear(self) -> None:
        pass  # No remote buffer to clear in-process

    async def send_dtmf(self, digit: str) -> None:
        if self._peer is not None and self._peer._inject is not None:
            from shuo.types import DTMFToneEvent
            self._peer._inject(DTMFToneEvent(digits=digit))

    async def hangup(self) -> None:
        if self._peer is not None and self._peer._on_stop is not None:
            await self._peer._on_stop()

    async def stop(self) -> None:
        if self._task is not None:
            await self._queue.put(None)  # sentinel
            await self._task
            self._task = None

    async def call(self, phone: str, twiml_url: str) -> None:
        pass  # Pairing happens at construction time via pair()
```

Key implementation details:
- `_reader()` background task reads from `self._queue` (unbounded asyncio.Queue, `await queue.put()` not `put_nowait`)
- `send_audio(payload: str)` decodes base64 to bytes before putting in peer's queue (on_media receives bytes)
- `send_dtmf` uses `self._peer._inject` which is a callable set externally (by conversation loop) — for tests, set it manually to a list.append or mock
- `hangup()` calls `peer._on_stop()` to signal stream end
- `stop()` sends None sentinel to own queue and awaits task completion
- `start()` fires `on_start` IMMEDIATELY after spawning the reader task with synthetic stream_sid `"local-{uuid_hex}"`, call_sid `"local-call-sid"`, phone `"local"`
