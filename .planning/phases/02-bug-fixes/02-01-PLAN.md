   ```python
   async def _evict_stale(self) -> None:
       """Remove connections that have been idle past TTL."""
       now = time.monotonic()
       stale: List[_Entry] = []
       async with self._lock:
           fresh: List[_Entry] = []
           for entry in self._ready:
               age = now - entry.created_at
               if age < self._ttl:
                   fresh.append(entry)
               else:
                   stale.append(entry)
           self._ready = fresh
       # Cancel outside lock — never hold lock during I/O
       for entry in stale:
           age_ms = int((now - entry.created_at) * 1000)
           log.info(f"Evicted stale connection (idle {age_ms}ms)")
           await entry.tts.cancel()
   ```

3. Rewrite `get` (lines 88-109) to hold lock only during list pop:
   ```python
   async def get(self, on_audio, on_done):
       """Get a connected TTS service with the given callbacks."""
       while True:
           async with self._lock:
               if self._ready:
                   entry = self._ready.pop(0)
               else:
                   entry = None
           if entry is None:
               break
           age = time.monotonic() - entry.created_at
           if age < self._ttl:
               entry.tts.bind(on_audio, on_done)
               age_ms = int(age * 1000)
               log.info(f"Dispensed warm connection (idle {age_ms}ms)")
               self._trigger_fill()
               return entry.tts
           else:
               age_ms = int(age * 1000)
               log.info(f"Discarded stale connection (idle {age_ms}ms)")
               await entry.tts.cancel()  # outside lock

       # No warm connections available -- create fresh (blocking)
       log.info("Pool empty, connecting fresh...")
       tts = create_tts(on_audio=on_audio, on_done=on_done)
       await tts.start()
       self._trigger_fill()
       return tts
   ```

   ```python
   async with self._lock:
       entries = list(self._ready)
       self._ready.clear()
   for entry in entries:
       await entry.tts.cancel()
   ```

CRITICAL: The lock is ONLY held during list reads/writes. All `await entry.tts.cancel()` calls happen OUTSIDE the lock to avoid serializing I/O.
